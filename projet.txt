//Initialiser GIT :
git Init

//Configurer les identifiants :
git config user.email ""

git add .

//Créer une version
git commit -m "Mon message"

//Aide sur GIT :
git help *command*

//Voir la liste des versions :
git log
git log --oneline

//regarder un commit précédent
git checkout "clé_commit"
